            $ vi ~/.vimrc 
            
            set number            " line 표시를 해줍니다.
            set ai                    " auto indent
            set si                    " smart indent
            set cindent            " c style indent
            set shiftwidth=4      " shift를 4칸으로 ( >, >>, <, << 등의 명령어)
            set tabstop=4         " tab을 4칸으로
            set ignorecase      " 검색시 대소문자 구별하지않음
            set hlsearch         " 검색시 하이라이트(색상 강조)
            set expandtab       " tab 대신 띄어쓰기로
            set background=dark  " 검정배경을 사용할 때, (이 색상에 맞춰 문법 하이라이트 색상이 달라집니다.)
            set nocompatible   " 방향키로 이동가능
            set fileencodings=utf-8,euc-kr    " 파일인코딩 형식 지정
            set bs=indent,eol,start    " backspace 키 사용 가능
            set history=1000    " 명령어에 대한 히스토리를 1000개까지
            set ruler              " 상태표시줄에 커서의 위치 표시
            set nobackup      " 백업파일을 만들지 않음
            set title               " 제목을 표시
            set showmatch    " 매칭되는 괄호를 보여줌
            set nowrap         " 자동 줄바꿈 하지 않음
            set wmnu           " tab 자동완성시 가능한 목록을 보여줌
            syntax on        " 문법 하이라이트 킴"
            
            # echo : 텍스트, 변수를 출력하는 명령어 
            # echo -e : escape 문자를 사용하게 하는 옵션
            # grep : 명령은 파일 내에서 지정한 패턴이나 문자열을 찾은 후에, 그 패턴을 포함하고 있는 모든 행을 표준 출력해 준다.
            $ grep [옵션] [찾을 문자열 정규 표현식] [파일명]
            
            # 출력 리다이렉션 - 명령 실행의 표준 출력을 파일로 저장합니다. 
            $echo "Hello, Seung Hwa" > hello.txt 
            
            $echo "Hello, Seung Hwa" >> hello.txt
            
            # 입력 리다이렉션 - 파일의 내용을 읽어 명령의 표준 입력으로 사용합니다. 
            $cat < hello.txt
            
            # 파이프, 명령 실행의 표준 출력을 다른 명령의 표준 입력으로 보냅니다. 
            $ls -al | grep .txt
            
            # Bash의 변수
            hello = "Hello World"
            echo $hello
            
            # 명령 실행 결과를 변수화 합니다. 
            $ docker rm $(docker ps -aq)
            $ echo $(date)
            
            # $()과 마찬가지로 명령 실행 결과를 변수화합니다. 
            $ docker rm `docker ps -aq`
            $ echo `date`
            
            # && 한 줄에서 명령을 여러개 실행합니다. 
            $ ls && ls -al
            
            # ; 한줄에서 명령을 여러 개 실행합니다. 
            $ false; echo "Hello"
            
            # ' ' 문자열 입니다. $USR 가 그대로 출력됩니다. 
            $ echo '$USER'
            
            # " " 문자열 입니다. 명령에 문자열 매개변수를 입력하거나 변수에 저장할 때 주로 사용합니다. 
            $ echo "Hello World"
            $ echo "$USER"
            $ echo "Host name is ${hostname}"
            $ echo "Time : `date`"
            
            # " " 안에 ' ' 가 들어갈 수 있습니다. 명령 안에서 다시 명령을 실행하고 매개 변수를 지정할 때 사용합니다. 
            $ bash -c "/bin/echo Hello 'World'" 
            $ bash -c "/bin/bash Hello 'World' >> hello.txt"
            
            # ' ' 안에서 "를 사용할 때는 \" 처럼 앞에 \를 붙여줍니다. 
            # " " 안에서 ". $ , ₩" 등의 특수문자를 그대로 사용하려면 앞에 \를 붙여줍니다. 
            $ bash -c "/bin/echo '{ \"user\": \"$USER\" }'"
            
            # 변수 치환(substitution)입니다. " " 문자열 안에서 변수를 출력할 때 주로 사용합니다. ${} 대신 $ 만 사용해도 됩니다. 
            $ str="World"
            $ echo "Hello ${str}"
            
            $ HELLO=
            $ HELLO=${HELLO-"abcd"}
            $ echo $HELLO
            
            $ WORLD=
            $ WORLD=${WORLD:-"abcd"}
            $ echo $WORLD
            
            # \ 한 줄로된 명령을 여러 줄로 표현할 때 사용합니다. 
            $ sudo docker run -d --name busybox:lastet 
            $ sudo docker run \
                          -d \
                          --name hello \
                          busybox:lastest
            
            # 연속된 숫자를 표현합니다. {시작숫자 ... 끝숫자} 형식입니다. 
            $ echo {1..10}
            
            # {} 안에 문자열을 여러 개 지정하여 명령 실행 횟수를 줄입니다. 다음은 hello.txt, world.txt 두 파일을 한 번에 hello-dir 디렉터리 아래에 복사합니다.
            $ cp ./{hello.txt,world.txt} hello-dir/
            
            # if 조건문입니다. 변수와 변수 끼리 또는 문자열과 비교할 때 사용합니다. 
            if [ $a -eq $b ]; then 
              echo $a
            fi
            
            숫자비교 
             -eq : 같다 
             -ne : 같지 않다
             -gt : 초과
             -ge : 이상
             -lt : 미만
             -le : 이하
            문자열 비교 
             =, == : 같다
             != : 같지 않다
             -z : 문자열이 NULL일 때
             -n : 문자열이 NULL이 아닐 때 
            
            # for 반복문 입니다. 변수 안에 있는 값을 반복하거나 범위를 지정하여 반복할 수 있습니다. 
            for i in ${ls}
            do 
              echo $i
            done
            
            for (( i=0; i < 10 ; i ++))
            do 
             echo $i
            done 
            
            NUM=(1, 2, 3)
            for i in ${NUM[@]}
            do
             echo $i
            done 
            
            # while 반복문 입니다. 
            while :
            do 
              echo "Hello World";
              sleep 1;
            done 
            
            # <<< 문자열 명령(프로세스)의 표준 입력으로 보냅니다. 
            $ cat <<< "User name is $USER"
            
            # <<EOF 여려줄의 문자열을 명령(프로세스)의 표준 입력으로 보냅니다. 
            
            cat> ./hello.txt <<EOF
            Hello World
            Host name is ${hostname}
            User name is ${USER}
            EOF
            
            # export 설정한 값을 환경 변수로 만듭니다. export <변수>=<값> 형식입니다. 
            $ export HELLO=world

Docker Publishing

서버 한 대에 애플리케이션 배포 하기

  • 개발자의 PC에서 애플리케이션을 개발한다.
  • git push 명령으로 소스를 서버에 올린다.
  • 서버에서는 저장소에 git push 명령이 발생하면 git hook 을 실행시킨다.
  • git hook 에서 Docker 이미지를 생성하고, 이미지를 컨테이너로 실행한다.
git config 명령으로 자신의 이메일과 이름을 설정한다.
git add, git commit 명령어로 개발자 pc의 exampleapp 저장소에 파일을 커밋합니다.
서버에서 Docker 이미지를 생성할 수 있도록 개발자 PC에서 DockerFile 을 작성합니다.
