A metadata facility
클래스, 인터페이스, 필드 그리고 메소드에 대해서 개별의 특성을 가질 수 있도록 만든다. metadata도 데이터에 대한 것이다. metadata는 실제 데이터에 flags를 추가하는 것으로, runtime 시점에 JVM이 해당 flags를 인식할 수 잇게 하여 context에 대해서 적절한 결정을 할 수 있도록 meta 정보를 조합할 수 있다.
  • 요소(Element)의 기본 값
    요소에 기본 값을 지정할 수 있다. class 에서 annotation을 선언하여 사용할 때, 해당 값이 선언되지 않으면 default에 있는 값을 입력한다. 
                                    
@interface HelloAnnotation {
  String value() default "hello my friends";
}
  • @Retention 
                                    
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
  
//@Retention(RetentionPolicy.CLASS)  /* 컴파일 이후에도 JVM에 의해서 참조가 가능합니다. */
@Retention(RetentionPolicy.RUNTIME)  /* 컴파일러가 클래스를 참조할 때까지 유효합니다. */ 
//@Retention(RetentionPolicy.SOURCE) /* 어노테이션 정보는 컴파일 이후 없어집니다. */
public @interface HelloAnnotation
{
    //some code
}
  • @Documented
    @Documented 어노테이션이 지정된 대상의 JavaDoc에 이 어노테이션의 존재를 표기하다록 지정 
                                    
import java.lang.annotation.Documented;

@Documented
public @interface HelloAnnotation {
    //Some other code
}
  • @Target
    해당 어노테이션이 사용하게 될 자바 요소를 지정할 수 있다.
    • ElementType.PACKAGE, // 패키지 선언시
    • ElementType.TYPE, // 타입 선언시
    • ElementType.CONSTRUCTOR, // 생성자 선언시
    • ElementType.FIELD, // 멤버 변수 선언시
    • ElementType.METHOD, // 메소드 선언시
    • ElementType.ANNOTATION_TYPE, // 어노테이션 타입 선언시
    • ElementType.LOCAL_VARIABLE, // 지역 변수 선언시
    • ElementType.PARAMETER, // 매개 변수 선언시
    • ElementType.TYPE_PARAMETER, // 매개 변수 타입 선언시
    • ElementType.TYPE_USE // 타입 사용시
                                    
import java.lang.annotation.ElementType;
import java.lang.annotation.Target;
  
@Target(value = {
  ElementType.TYPE, 
  ElementType.METHOD, 
  ElementType.CONSTRUCTOR, 
  ElementType.ANNOTATION_TYPE, 
  ElementType.FIELD, 
  ElementType.LOCAL_VARIABLE,                
  ElementType.PACKAGE, 
  ElementType.PARAMETER
})
public @interface HelloAnnotation {
    //Some other code
}
  • @Inherited
    이 어노테이션을 사용한 슈퍼 클래스를 상속한 서브 클래스에서도 해당 어노테이션을 갖도록 한다. 
                                    
import java.lang.annotation.Inherited;

@Inherited
public @interface HelloAnnotation {
    //Some other code
}
